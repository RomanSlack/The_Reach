  1. Start Postgres:
  docker-compose up -d

  2. Start Backend:
  cd backend
  python -m venv .venv && source .venv/bin/activate
  pip install -r requirements.txt
  uvicorn main:app --reload

  3. Start Frontend:
  cd frontend
  npm run dev

  4. (Optional) Desktop App:
  cd frontend
  npm run tauri dev
